import akka.actor.{Actor, ActorSystem, Props}

// Define the SwitchActor
class SwitchActor extends Actor {
  var state: String = "happy"

  def receive: Receive = {
    case "sad" =>
      state = "sad"
      println(s"SwitchActor switched to '$state' state.")
    case msg =>
      println(s"SwitchActor received: $msg. State is '$state'.")
      println(s"I'm $state!")
  }
}

// Usage Example (for testing purposes)
object SwitchApp extends App {
  val system = ActorSystem("SwitchSystem")
  val switchActor = system.actorOf(Props[SwitchActor](), "switchActor")

  // Send messages to toggle states and demonstrate responses
  switchActor! "sad"
  switchActor! "happy"
  switchActor! "happy"
  switchActor! "sad"
  switchActor! "happy"

  // Allow some time for processing before shutdown
  Thread.sleep(1000)
  system.terminate()
}

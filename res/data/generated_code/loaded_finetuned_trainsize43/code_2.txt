import akka.actor.{Actor, ActorRef, ActorSystem, Props}

// Define the HelloWorldActor
class HelloWorldActorsender extends Actor {
  def receive: Receive = {
    case "hello" => sender()! "Hello!"
    case msg       => println(s"Received message: $msg")
  }
}

// Usage Example
object HelloWorldApp extends App {
  val system = ActorSystem("HelloWorldSystem")
  
  val helloWorld = system.actorOf(Props[HelloWorldActorsender](), "helloWorldActor")
  
  helloWorld! "hello"
}

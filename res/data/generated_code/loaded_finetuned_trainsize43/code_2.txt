import akka.actor.{Actor, ActorSystem, Props}

// Define the HelloWorldActor
class HelloWorldActor extends Actor {
  def apply(): Unit = {
    println("Hello from Akka Actor!")
  }
}

// Usage Example
object HelloWorldApp extends App {
  val system = ActorSystem("HelloWorldSystem")
  val helloActor = system.actorOf(Props(new HelloWorldActor()), "helloActor")

  // Send a message to the actor
  helloActor! ()

  // Allow some time for processing before shutdown
  Thread.sleep(1000)
  system.terminate()
}
